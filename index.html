def factorial(n): """Calculate the factorial of a number.""" if n < 0: return "Invalid input. Factorial is not defined for negative numbers." elif n == 0 or n == 1: return 1 else: result = 1 for i in range(2, n + 1): result *= i return result def is_prime(num): """Check if a number is a prime number.""" if num <= 1: return False for i in range(2, int(num ** 0.5) + 1): if num % i == 0: return False return True def check_prime_numbers(numbers): """Check and return prime numbers from a list.""" prime_numbers = [num for num in numbers if is_prime(num)] return prime_numbers # Example usage number = 5 print(f"The factorial of {number} is {factorial(number)}.") numbers_list = [2, 3, 4, 5, 10, 13, 17, 20] prime_numbers = check_prime_numbers(numbers_list) print(f"Prime numbers in the list are: {prime_numbers}.")
